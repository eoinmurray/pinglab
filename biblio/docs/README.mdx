```python
import pinglab

    run_network, 
    types, 
    inputs, 
    analysis, 
    plots, 
    utils, 
    multiprocessing
)
```

## Quick Start

```python
import numpy as np
from pinglab import run_network
from pinglab.types import NetworkConfig
from pinglab.inputs import tonic

# Configure network
config = NetworkConfig(
    N_E=800,              # Excitatory neurons
    N_I=200,              # Inhibitory neurons
    T=1000.0,             # Simulation duration (ms)
    dt=0.1,               # Time step (ms)
    g_ei=0.5,             # E→I coupling strength
    g_ie=1.0,             # I→E coupling strength
)

# Generate external input
num_steps = int(config.T / config.dt)
external_input = tonic(
    N_E=config.N_E,
    N_I=config.N_I,
    I_E=1.5,              # Current to E neurons
    I_I=1.0,              # Current to I neurons
    noise_std=0.3,        # Gaussian noise
    num_steps=num_steps,
    seed=42,
)

# Run simulation
result = run_network(config, external_input)

# Analyze results
from pinglab.analysis import mean_firing_rates
E_rate, I_rate = mean_firing_rates(result.spikes, config.N_E, config.N_I)
print(f"E: {E_rate:.1f} Hz, I: {I_rate:.1f} Hz")
```

## Core Function

### `run_network`

```python
from pinglab import run_network
from pinglab.types import NetworkConfig, NetworkResult

result: NetworkResult = run_network(config: NetworkConfig, external_input: np.ndarray)
```

Runs a conductance-based leaky integrate-and-fire (LIF) network simulation.

**Network Features:**
- Excitatory (E) and inhibitory (I) populations
- Exponentially decaying AMPA and GABA conductances
- Configurable synaptic delays via ring buffers
- Optional per-neuron heterogeneity
- Optional instrument recording (voltage, conductance traces)

**Parameters:**
- `config` (`NetworkConfig`): Network configuration
- `external_input` (`np.ndarray`): Shape `(num_steps, N)` or `(num_steps,)` for uniform input

**Returns:** `NetworkResult` containing spike data and optional instrument recordings

## Types

All types are Pydantic models with validation.

### `NetworkConfig`

Main simulation configuration.

```python
from pinglab.types import NetworkConfig

config = NetworkConfig(
    # Required
    dt=0.1,                    # Time step (ms)
    T=1000.0,                  # Total simulation time (ms)
    N_E=800,                   # Number of excitatory neurons
    N_I=200,                   # Number of inhibitory neurons
    g_ei=0.5,                  # E→I coupling strength
    g_ie=1.0,                  # I→E coupling strength

    # Optional - LIF parameters
    V_init=-65.0,              # Initial membrane potential (mV)
    E_L=-65.0,                 # Leak reversal potential (mV)
    E_e=0.0,                   # Excitatory reversal potential (mV)
    E_i=-80.0,                 # Inhibitory reversal potential (mV)
    C_m_E=1.0,                 # E membrane capacitance (nF)
    C_m_I=1.0,                 # I membrane capacitance (nF)
    g_L_E=0.1,                 # E leak conductance (μS)
    g_L_I=0.1,                 # I leak conductance (μS)
    V_th=-50.0,                # Spike threshold (mV)
    V_reset=-65.0,             # Reset potential (mV)

    # Synaptic time constants
    tau_ampa=5.0,              # AMPA decay time constant (ms)
    tau_gaba=10.0,             # GABA decay time constant (ms)

    # Refractory periods
    t_ref_E=3.0,               # E refractory period (ms)
    t_ref_I=1.5,               # I refractory period (ms)

    # Synaptic delays
    delay_ei=1.5,              # E→I delay (ms)
    delay_ie=1.5,              # I→E delay (ms)
    delay_ee=1.5,              # E→E delay (ms)
    delay_ii=1.5,              # I→I delay (ms)

    # Intra-population coupling
    g_ee=0.0,                  # E→E coupling strength
    g_ii=0.0,                  # I→I coupling strength

    # Connection probabilities
    p_ee=1.0,                  # E→E connection probability
    p_ei=1.0,                  # E→I connection probability
    p_ie=1.0,                  # I→E connection probability
    p_ii=1.0,                  # I→I connection probability

    # Connectivity scaling
    connectivity_scaling="one_over_N_src",  # or "none"

    # Heterogeneity (standard deviations)
    V_th_heterogeneity_sd=0.0,
    g_L_heterogeneity_sd=0.0,
    C_m_heterogeneity_sd=0.0,
    t_ref_heterogeneity_sd=0.0,

    # Random seed
    seed=42,

    # Instrument recording (optional)
    instruments=None,          # InstrumentsConfig or None
)
```

### `Spikes`

Spike data container.

```python
from pinglab.types import Spikes

spikes = Spikes(
    times=np.array([...]),     # Spike times in ms
    ids=np.array([...]),       # Neuron indices
    types=np.array([...]),     # 0=E, 1=I (optional)
    populations=None,          # For multi-population (optional)
)
```

### `NetworkResult`

Simulation output.

```python
from pinglab.types import NetworkResult

result.spikes          # Spikes object
result.instruments     # InstrumentsResults or None
```

### `InstrumentsConfig`

Configuration for recording state variables.

```python
from pinglab.types import InstrumentsConfig

instruments = InstrumentsConfig(
    variables=['V', 'g_e', 'g_i'],  # Variables to record
    neuron_ids=[0, 1, 100],         # Neurons to record from
    downsample=10,                   # Record every Nth step
    population_means=True,           # Also record population averages
)
```

### `InstrumentsResults`

Recorded state variable traces.

```python
result.instruments.times       # Time points (ms)
result.instruments.neuron_ids  # Recorded neuron IDs
result.instruments.types       # Neuron types (0=E, 1=I)
result.instruments.V           # Membrane potential traces
result.instruments.g_e         # Excitatory conductance traces
result.instruments.g_i         # Inhibitory conductance traces

# Population means (if population_means=True)
result.instruments.V_mean_E    # Mean E population voltage
result.instruments.V_mean_I    # Mean I population voltage
result.instruments.g_e_mean_E  # Mean E excitatory conductance
result.instruments.g_e_mean_I  # Mean I excitatory conductance
result.instruments.g_i_mean_E  # Mean E inhibitory conductance
result.instruments.g_i_mean_I  # Mean I inhibitory conductance
```

## Input Generation

### `inputs.tonic`

Constant baseline input with Gaussian noise.

```python
from pinglab.inputs import tonic

external_input = tonic(
    N_E=800,           # Number of E neurons
    N_I=200,           # Number of I neurons
    I_E=1.5,           # Baseline current to E neurons
    I_I=1.0,           # Baseline current to I neurons
    noise_std=0.3,     # Noise standard deviation
    num_steps=10000,   # Simulation steps
    seed=42,           # Random seed
)
# Returns: np.ndarray of shape (num_steps, N_E + N_I)
```

### `inputs.oscillating`

Sinusoidal input with noise.

```python
from pinglab.inputs import oscillating

external_input = oscillating(
    N_E=800,
    N_I=200,
    I_E=1.5,
    I_I=1.0,
    noise_std=0.3,
    num_steps=10000,
    dt=0.1,                      # Time step (ms)
    seed=42,
    oscillation_freq=40.0,       # Frequency (Hz)
    oscillation_amplitude=0.5,   # Amplitude
    oscillation_phase=0.0,       # Initial phase (radians)
    phase_offset_I=0.0,          # I phase offset from E (radians)
)
```

### `inputs.add_pulse_to_input`

Add a pulse stimulus to specific neurons.

```python
from pinglab.inputs import add_pulse_to_input

modified_input = add_pulse_to_input(
    baseline_input=external_input,
    target_neurons=np.arange(100),  # First 100 neurons
    pulse_t=500.0,                   # Pulse onset (ms)
    pulse_width_ms=10.0,             # Pulse duration (ms)
    pulse_amp=2.0,                   # Amplitude to add
    dt=0.1,                          # Time step (ms)
)
```

### `inputs.compute_spike_delta`

Measure spike count change around a pulse.

```python
from pinglab.inputs import compute_spike_delta

delta = compute_spike_delta(
    spikes=result.spikes,
    target_neurons=np.arange(100),
    pulse_t=500.0,
    pre_window_ms=50.0,    # Window before pulse
    post_window_ms=50.0,   # Window after pulse
)
# Returns: int (post_spikes - pre_spikes)
```

## Analysis

### `analysis.mean_firing_rates`

Compute mean firing rates for E and I populations.

```python
from pinglab.analysis import mean_firing_rates

E_rate, I_rate = mean_firing_rates(
    spikes=result.spikes,
    N_E=800,
    N_I=200,
)
# Returns: (float, float) in Hz
```

### `analysis.population_rate`

Compute time-binned population firing rate.

```python
from pinglab.analysis import population_rate

t_ms, rate_hz = population_rate(
    spikes=result.spikes,
    T_ms=1000.0,           # Total simulation time
    dt_ms=1.0,             # Bin width
    pop="E",               # "E", "I", or "all"
    N_E=800,
    N_I=200,
)
# Returns: (bin_centers in ms, firing rate in Hz)
```

### `analysis.rate_psd`

Power spectral density of population rate.

```python
from pinglab.analysis import rate_psd

freqs, psd = rate_psd(
    rate_hz=rate_hz,       # From population_rate
    dt_ms=1.0,             # Bin width used
    nperseg=1024,          # Welch segment length (optional)
)
# Returns: (frequencies in Hz, power spectral density)
```

### `analysis.crosscorr`

E→I cross-correlation histogram.

```python
from pinglab.analysis import crosscorr

centers, hist = crosscorr(
    spikes=result.spikes,
    N_E=800,
    bin_ms=1.0,            # Histogram bin width
    max_lag_ms=20.0,       # Maximum lag to consider
)
# Returns: (bin centers in ms, spike counts per bin)
```

### `analysis.population_isi_cv`

Inter-spike interval coefficient of variation.

```python
from pinglab.analysis import population_isi_cv

cv_E, cv_I = population_isi_cv(
    spikes=result.spikes,
    N_E=800,
    N_I=200,
    min_spikes=2,          # Minimum spikes per neuron
)
# Returns: (float or None, float or None) - median CV across neurons
```

## Plotting

All plotting functions save both light and dark theme versions (`_light.png` and `_dark.png`).

### `plots.save_raster`

Save spike raster plot.

```python
from pinglab.plots import save_raster
from pathlib import Path

save_raster(
    spikes=result.spikes,
    path=Path("output/raster.png"),
    label="Simulation",
    vertical_lines=[500.0],        # Optional: mark times
    external_input=external_input, # Optional: show input trace
    dt=0.1,                        # Required if external_input provided
)
# Creates: output/raster_light.png, output/raster_dark.png
```

### `plots.save_instrument_traces`

Save instrument recording traces.

```python
from pinglab.plots import save_instrument_traces
from pathlib import Path

save_instrument_traces(
    instruments=result.instruments,
    save_path=Path("output/"),
)
# Creates: trace_voltage_light.png, trace_voltage_dark.png, etc.
```

### `plots.styles`

Theme styles for custom plots.

```python
from pinglab.plots.styles import apply_style, save_both, LIGHT, DARK

# Apply a style
apply_style(LIGHT)  # or DARK

# Save both light and dark versions
def my_plot():
    plt.figure()
    plt.plot(x, y)
    plt.xlabel("Time")

save_both("output/custom.png", my_plot)
```

## Utilities

### `utils.slice_spikes`

Extract spikes within a time window.

```python
from pinglab.utils import slice_spikes

sliced = slice_spikes(
    spikes=result.spikes,
    start_time=500.0,      # Start time (inclusive)
    stop_time=1000.0,      # Stop time (exclusive)
)
# Returns: new Spikes object with only spikes in window
```

### `utils.expand_parameter_spec`

Expand parameter sweeps into configurations.

```python
from pinglab.utils import expand_parameter_spec

spec = {
    "g_ei": {"type": "linspace", "start": 0.1, "stop": 1.0, "num": 10},
    "g_ie": {"type": "values", "values": [0.5, 1.0, 2.0]},
}

param_dicts = expand_parameter_spec(spec, mode="grid")
# Returns: list of dicts with all combinations
# [{"g_ei": 0.1, "g_ie": 0.5}, {"g_ei": 0.1, "g_ie": 1.0}, ...]
```

**Spec types:**
- `linspace`: `{"type": "linspace", "start": 0, "stop": 1, "num": 10}`
- `range`: `{"type": "range", "start": 0, "stop": 10, "step": 1}`
- `values`: `{"type": "values", "values": [1, 2, 3]}`

### `utils.load_config`

Load experiment config from YAML.

```python
from pinglab.utils import load_config
from pathlib import Path

config = load_config(Path("experiment.yaml"))
# Returns: ExperimentConfig
```

## Multiprocessing

### `multiprocessing.parallel`

Execute function in parallel over configurations.

```python
from pinglab.multiprocessing import parallel

def run_one(cfg: dict):
    config = NetworkConfig(**cfg)
    # ... generate input, run simulation
    return result

configs = [{"N_E": 800, "N_I": 200, ...}, ...]
results = parallel(run_one, configs, label="simulations")
# Returns: list of results with progress bar
```

Uses `joblib.Parallel` with all available cores.

## Example: Parameter Sweep

```python
from pinglab import run_network
from pinglab.types import NetworkConfig
from pinglab.inputs import tonic
from pinglab.analysis import mean_firing_rates
from pinglab.utils import expand_parameter_spec
from pinglab.multiprocessing import parallel

# Define parameter sweep
spec = {
    "g_ei": {"type": "linspace", "start": 0.2, "stop": 1.0, "num": 5},
    "g_ie": {"type": "linspace", "start": 0.5, "stop": 2.0, "num": 5},
}
param_dicts = expand_parameter_spec(spec, mode="grid")

def run_one(params):
    config = NetworkConfig(
        N_E=800, N_I=200, T=1000.0, dt=0.1,
        **params
    )
    num_steps = int(config.T / config.dt)
    ext_input = tonic(config.N_E, config.N_I, 1.5, 1.0, 0.3, num_steps, 42)
    result = run_network(config, ext_input)
    E_rate, I_rate = mean_firing_rates(result.spikes, config.N_E, config.N_I)
    return {"params": params, "E_rate": E_rate, "I_rate": I_rate}

results = parallel(run_one, param_dicts, label="sweep")
```

## Example: Recording Instruments

```python
from pinglab import run_network
from pinglab.types import NetworkConfig, InstrumentsConfig
from pinglab.inputs import tonic
from pinglab.plots import save_instrument_traces
from pathlib import Path

config = NetworkConfig(
    N_E=800, N_I=200, T=500.0, dt=0.1,
    g_ei=0.5, g_ie=1.0,
    instruments=InstrumentsConfig(
        variables=['V', 'g_e', 'g_i'],
        neuron_ids=[0, 1, 800, 801],  # 2 E neurons, 2 I neurons
        downsample=10,
        population_means=True,
    ),
)

num_steps = int(config.T / config.dt)
ext_input = tonic(config.N_E, config.N_I, 1.5, 1.0, 0.3, num_steps, 42)
result = run_network(config, ext_input)

# Plot traces
save_instrument_traces(result.instruments, Path("output/"))
```

## Numerical Considerations

**Time step requirements:**
- The simulation uses forward Euler integration
- Requirement: $dt < \tau_{mem} / 5$ where $\tau_{mem} = C_m / g_L$
- Warning issued if $dt > \tau_{mem} / 10$
- Default parameters: $\tau_{mem} = 10$ ms, so $dt \leq 1$ ms recommended

**Firing rate limits:**
- Maximum E rate: $1000 / t_{ref,E}$ Hz (default: 333 Hz)
- Maximum I rate: $1000 / t_{ref,I}$ Hz (default: 667 Hz)
- Warnings issued if rates exceed theoretical maximums
